ПРОТОКОЛ УПРАВЛЕНИЯ ЧАЙНИКОМ

-- КЛИЕНТ --
H 	- нагрев (heat)
R<id> 	- запрос чтения показаний датчика с номером = id (read)
K 	- выключение (kill)
A<ssid>;<password>;	- запрос подключения чайника к внешней точке доступа (ap)
-- -- -- --


-- СЕРВЕР --
T<id><value>;\n	- отправка показаний (value) датчика с номером = id (transmit)
E<id>;\n	- сообщение об ошибке с номером id (error)
D;\n		- сообщение о вскипевшей воде (done)
H;\n		- подтверждение включения нагрева
K;\n		- подтверждение отключения
OK;\n		- подтверждение подключения к точке доступа
IP<ip>;n	- сообщение своего IP в сети
-- -- -- --

-- id датчиков --
0	- состояние (вкл/выкл)
1-3	- [raw] датчики давления (1 - перед, 2 - левый, 3 - правый)
4 	- [raw] датчик температуры 
5 	- количество воды (единица измерения - 0.1л; пример: значение 15 = 1.5л)
6 	- температура в градусах цельсия
-- -- -- -- -- --

-- коды ошибок --
0	- none
1 	- низкий уровень воды
2	- высокий уровень воды
3	- датчик температуры сломан
4	- открыта крышка
5-7	- сломан датчик давления (5 - перед, 6 - левый, 7 - правый)
8	- нет ответа (bluetooth)
9	- нет ответа (Wi-Fi)
10	- id не распознан
11 	- команда не распознана
12	- запрашиваемая функция не реализована
13	- ошибка контрольной суммы
14	- не удалось подключиться к точке доступа
